42 void EMCascade::process(Candidate *candidate) const {
43  int id = candidate->current.getId();
44  if ((id != 22) and (id != 11) and (id != -11))
45  return;
46 
47  candidate->setActive(false);
48 
49  double logE = log10(candidate->current.getEnergy() / eV);
50  double D = candidate->current.getPosition().getR(); // distance to (0,0,0)
51 
52  if ((logE < logEmin) or (logE > logEmax))
53  return;
54  if (D > Dmax)
55  return;
56 
57  int iE = (logE - logEmin) / dlogE;
58  int iD = D / dD;
59  int i = (iD * nE) + iE;
60 
61 #pragma omp critical
62  {
63  if (id == 22)
64  photonHist[i] += 1;
65  else if (id == 11)
66  electronHist[i] += 1;
67  else
68  positronHist[i] += 1;
69  }
70 }

114 void EMCascade::runCascade(const std::string &filename, int IRBFlag,
115  int RadioFlag, double Bfield, double cutCascade) {
116 
117  // set up DINT
118  std::string dataPath = getDataPath("dint");
119  double B = Bfield / gauss;
120  double h = H0() * Mpc / 1000;
121  DintEMCascade dint(IRBFlag, RadioFlag, dataPath, B, h, omegaM(), omegaL());
122 
123  Spectrum inputSpectrum, outputSpectrum;
124  NewSpectrum(&inputSpectrum, nE);
125  NewSpectrum(&outputSpectrum, nE);
126  InitializeSpectrum(&outputSpectrum);
127 
128  // step-wise cascade calculation
129  for (int iD = nD - 1; iD >= 0; iD--) {
130  // make output of previous step the new input and reset output
131  SetSpectrum(&inputSpectrum, &outputSpectrum);
132  InitializeSpectrum(&outputSpectrum);
133 
134  // add new particles to input
135  double count = 0;
136  for (int iE = 0; iE < nE; iE++) {
137  int i = (iD * nE) + iE;
138  inputSpectrum.spectrum[PHOTON][iE] += photonHist[i];
139  inputSpectrum.spectrum[ELECTRON][iE] += electronHist[i];
140  inputSpectrum.spectrum[POSITRON][iE] += positronHist[i];
141  count += inputSpectrum.spectrum[PHOTON][iE];
142  count += inputSpectrum.spectrum[ELECTRON][iE];
143  count += inputSpectrum.spectrum[POSITRON][iE];
144  }
145  // skip step if input spectrum empty
146  if (count == 0)
147  continue;
148 
149  // start and stop distance [Mpc,light travel] from bin center to bin center
150  double D1 = comoving2LightTravelDistance( (iD + 0.5) * dD );
151  double D0 = comoving2LightTravelDistance( std::max((iD - 0.5) * dD, 0.) );
152 
153  // propagate distance step
154  dint.propagate(D1/Mpc, D0/Mpc, &inputSpectrum, &outputSpectrum, cutCascade);
155  }
156 
157  // write output
158  std::ofstream outfile(filename.c_str());
159  if (!outfile) {
160  std::stringstream s;
161  s << "EMCascade: could not open " << filename;
162  throw std::runtime_error(s.str());
163  }
164  outfile << "# log10(E/eV) photons electrons positrons\n";
165  for (int iE = 0; iE < nE; iE++) {
166  outfile << std::setw(5) << logEmin + (iE + 0.5) * dlogE;
167  for (int s = 0; s < 3; s++)
168  outfile << std::setw(13) << outputSpectrum.spectrum[s][iE];
169  outfile << "\n";
170  }
171  outfile.close();
172 
173  // clear the histogram
174  photonHist.assign(nD * nE, 0);
175  electronHist.assign(nD * nE, 0);
176  positronHist.assign(nD * nE, 0);
177 
178  DeleteSpectrum(&outputSpectrum);
179  DeleteSpectrum(&inputSpectrum);
180 }
