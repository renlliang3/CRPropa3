28 void PropagationCK::tryStep(const Y &y, Y &out, Y &error, double h,
29  ParticleState &particle, double z) const {
30  std::vector<Y> k;
31  k.reserve(6);
32 
33  out = y;
34  error = Y(0);
35 
36  // calculate the sum of b_i * k_i
37  for (size_t i = 0; i < 6; i++) {
38 
39  Y y_n = y;
40  for (size_t j = 0; j < i; j++)
41  y_n += k[j] * a[i * 6 + j] * h;
42 
43  // update k_i
44  k[i] = dYdt(y_n, particle, z);
45 
46  out += k[i] * b[i] * h;
47  error += k[i] * (b[i] - bs[i]) * h;
48  }
49 }

51 PropagationCK::Y PropagationCK::dYdt(const Y &y, ParticleState &p, double z) const {
52  // normalize direction vector to prevent numerical losses
53  Vector3d velocity = y.u.getUnitVector() * c_light;
54  Vector3d B(0, 0, 0);
55  try {
56  B = field->getField(y.x, z);
57  } catch (std::exception &e) {
58  std::cerr << "PropagationCK: Exception in getField." << std::endl;
59  std::cerr << e.what() << std::endl;
60  }
61  // Lorentz force: du/dt = q*c/E * (v x B)
62  Vector3d dudt = p.getCharge() * c_light / p.getEnergy() * velocity.cross(B);
63  return Y(velocity, dudt);
64 }

80 void PropagationCK::process(Candidate *candidate) const {
81  // save the new previous particle state
82  ParticleState &current = candidate->current;
83  candidate->previous = current;
84 
85  double step = clip(candidate->getNextStep(), minStep, maxStep);
86 
87  // rectilinear propagation for neutral particles
88  if (current.getCharge() == 0) {
89  Vector3d pos = current.getPosition();
90  Vector3d dir = current.getDirection();
91  current.setPosition(pos + dir * step);
92  candidate->setCurrentStep(step);
93  candidate->setNextStep(maxStep);
94  return;
95  }
96 
97  Y yIn(current.getPosition(), current.getDirection());
98  Y yOut, yErr;
99  double newStep = step;
100  double r = 42; // arbitrary value > 1
101  double z = candidate->getRedshift();
102 
103  // try performing step until the target error (tolerance) or the minimum step size has been reached
104  while (r > 1) {
105  step = newStep;
106  tryStep(yIn, yOut, yErr, step / c_light, current, z);
107 
108  r = yErr.u.getR() / tolerance; // ratio of absolute direction error and tolerance
109  newStep = step * 0.95 * pow(r, -0.2); // update step size to keep error close to tolerance
110  newStep = clip(newStep, 0.1 * step, 5 * step); // limit the step size change
111  newStep = clip(newStep, minStep, maxStep);
112 
113  if (step == minStep)
114  break; // performed step already at the minimum
115  }
116 
117  current.setPosition(yOut.x);
118  current.setDirection(yOut.u.getUnitVector());
119  candidate->setCurrentStep(step);
120  candidate->setNextStep(newStep);
121 }
