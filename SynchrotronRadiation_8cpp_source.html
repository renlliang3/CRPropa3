94 void SynchrotronRadiation::process(Candidate *candidate) const {
95  double charge = fabs(candidate->current.getCharge());
96  if (charge == 0)
97  return; // only charged particles
98 
99  // calculate gyroradius, evaluated at the current position
100  double z = candidate->getRedshift();
101  double B;
102  if (field.valid()) {
103  Vector3d Bvec = field->getField(candidate->current.getPosition(), z);
104  B = Bvec.cross(candidate->current.getDirection()).getR();
105  } else {
106  B = sqrt(2. / 3) * Brms; // average perpendicular field component
107  }
108  B *= pow(1 + z, 2); // cosmological scaling
109  double Rg = candidate->current.getMomentum().getR() / charge / B;
110 
111  // calculate energy loss
112  double lf = candidate->current.getLorentzFactor();
113  double dEdx = 1. / 6 / M_PI / epsilon0 * pow(lf * lf - 1, 2) * pow(eplus / Rg, 2); // Jackson p. 770 (14.31)
114  double step = candidate->getCurrentStep() / (1 + z); // step size in local frame
115  double dE = step * dEdx;
116 
117  // apply energy loss and limit next step
118  double E = candidate->current.getEnergy();
119  candidate->current.setEnergy(E - dE);
120  candidate->limitNextStep(limit * E / dEdx);
121 
122  // optionally add secondary photons
123  if (not(havePhotons))
124  return;
125 
126  // check if photons with energies > 14 * Ecrit are possible
127  double Ecrit = 3. / 4 * h_planck / M_PI * c_light * pow(lf, 3) / Rg;
128  if (14 * Ecrit < secondaryThreshold)
129  return;
130 
131  // draw photons up to the total energy loss
132  Random &random = Random::instance();
133  while (dE > 0) {
134  // draw random value between 0 and maximum of corresponding cdf
135  // choose bin of s where cdf(x) = cdf_rand -> x_rand
136  size_t i = random.randBin(tabCDF); // draw random bin (upper bin boundary returned)
137  double binWidth = (tabx[i] - tabx[i-1]);
138  double x = tabx[i-1] + random.rand() * binWidth; // draw random x uniformly distributed in bin
139  double Egamma = x * Ecrit;
140 
141  // if the remaining energy is not sufficient check for random accepting
142  if (Egamma > dE)
143  if (random.rand() > (dE / Egamma))
144  break; // not accepted
145 
146  // create synchrotron photon and repeat with remaining energy
147  dE -= Egamma;
148  Vector3d pos = random.randomInterpolatedPosition(candidate->previous.getPosition(), candidate->current.getPosition());
149  if (Egamma > secondaryThreshold) // create only photons with energies above threshold
150  candidate->addSecondary(22, Egamma, pos);
151  }
152 }
