98 class PPSecondariesEnergyDistribution {
99  private:
100  std::vector<double> tab_s;
101  std::vector< std::vector<double> > data;
102  size_t N;
103 
104  public:
105  // differential cross section for pair production for x = Epositron/Egamma, compare Lee 96 arXiv:9604098
106  double dSigmadE_PPx(double x, double beta) {
107  double A = (x / (1. - x) + (1. - x) / x );
108  double B = (1. / x + 1. / (1. - x) );
109  double y = (1 - beta * beta);
110  return A + y * B - y * y / 4 * B * B;
111  }
112 
113  PPSecondariesEnergyDistribution() {
114  N = 1000;
115  size_t Ns = 1000;
116  double s_min = 4 * mec2 * mec2;
117  double s_max = 1e23 * eV * eV;
118  double dls = log(s_max / s_min) / Ns;
119  data = std::vector< std::vector<double> >(Ns, std::vector<double>(N));
120  tab_s = std::vector<double>(Ns + 1);
121 
122  for (size_t i = 0; i < Ns + 1; ++i)
123  tab_s[i] = s_min * exp(i*dls); // tabulate s bin borders
124 
125  for (size_t i = 0; i < Ns; i++) {
126  double s = s_min * exp(i*dls + 0.5*dls);
127  double beta = sqrt(1 - s_min/s);
128  double x0 = (1 - beta) / 2;
129  double dx = log((1 + beta) / (1 - beta)) / N;
130 
131  // cumulative midpoint integration
132  std::vector<double> data_i(1000);
133  data_i[0] = dSigmadE_PPx(x0, beta) * expm1(dx);
134  for (size_t j = 1; j < N; j++) {
135  double x = x0 * exp(j*dx + 0.5*dx);
136  double binWidth = exp((j+1)*dx)-exp(j*dx);
137  data_i[j] = dSigmadE_PPx(x, beta) * binWidth + data_i[j-1];
138  }
139  data[i] = data_i;
140  }
141  }

143  // sample positron energy from cdf(E, s_kin)
144  double sample(double E0, double s) {
145  // get distribution for given s
146  size_t idx = std::lower_bound(tab_s.begin(), tab_s.end(), s) - tab_s.begin();
147  std::vector<double> s0 = data[idx];
148 
149  // draw random bin
150  Random &random = Random::instance();
151  size_t j = random.randBin(s0) + 1;
152 
153  double s_min = 4 * mec2 * mec2;
154  double beta = sqrt(1 - s_min / s);
155  double x0 = (1 - beta) / 2.;
156  double dx = log((1 + beta) / (1 - beta)) / N;
157  double binWidth = x0 * (exp(j*dx) - exp((j-1)*dx));
158  if (random.rand() < 0.5)
159  return E0 * (x0 * exp((j-1) * dx) + binWidth);
160  else
161  return E0 * (1 - (x0 * exp((j-1) * dx) + binWidth));
162  }

165 void EMPairProduction::performInteraction(Candidate *candidate) const {
166  // scale particle energy instead of background photon energy
167  double z = candidate->getRedshift();
168  double E = candidate->current.getEnergy() * (1 + z);
169 
170  // cosmic ray photon is lost after interacting
171  candidate->setActive(false);
172 
173  // check if secondary electron pair needs to be produced
174  if (not haveElectrons)
175  return;
176 
177  // check if in tabulated energy range
178  if (E < tabE.front() or (E > tabE.back()))
179  return;
180 
181  // sample the value of s
182  Random &random = Random::instance();
183  size_t i = closestIndex(E, tabE); // find closest tabulation point
184  size_t j = random.randBin(tabCDF[i]);
185  double lo = std::max(4 * mec2 * mec2, tabs[j-1]); // first s-tabulation point below min(s_kin) = (2 me c^2)^2; ensure physical value
186  double hi = tabs[j];
187  double s = lo + random.rand() * (hi - lo);
188 
189  // sample electron / positron energy
190  static PPSecondariesEnergyDistribution interpolation;
191  double Ee = interpolation.sample(E, s);
192  double Ep = E - Ee;
193 
194  // sample random position along current step
195  Vector3d pos = random.randomInterpolatedPosition(candidate->previous.getPosition(), candidate->current.getPosition());
196  candidate->addSecondary(-11, Ee / (1 + z), pos);
197  candidate->addSecondary(11, Ep / (1 + z), pos);
198 }

200 void EMPairProduction::process(Candidate *candidate) const {
201  // check if photon
202  if (candidate->current.getId() != 22)
203  return;
204 
205  // scale particle energy instead of background photon energy
206  double z = candidate->getRedshift();
207  double E = candidate->current.getEnergy() * (1 + z);
208 
209  // check if in tabulated energy range
210  if (E < tabEnergy.front() or (E > tabEnergy.back()))
211  return;
212 
213  // interaction rate
214  double rate = interpolate(E, tabEnergy, tabRate);
215  rate *= pow(1 + z, 2) * photonFieldScaling(photonField, z);
216 
217  // check for interaction
218  Random &random = Random::instance();
219  double randDistance = -log(random.rand()) / rate;
220  if (candidate->getCurrentStep() > randDistance)
221  performInteraction(candidate);
222  else

224 }
