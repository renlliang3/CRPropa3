160 void SourceComposition::add(int id, double weight) {
161  nuclei.push_back(id);
162  int A = massNumber(id);
163  int Z = chargeNumber(id);
164 
165  double a = 1 + index;
166  if (std::abs(a) < std::numeric_limits<double>::min())
167  weight *= log(Z * Rmax / Emin);
168  else
169  weight *= (pow(Z * Rmax, a) - pow(Emin, a)) / a;
170 
171  weight *= pow(A, -a);
172 
173  if (cdf.size() > 0)
174  weight += cdf.back();
175  cdf.push_back(weight);
176  setDescription();

541  return theta;

857 void SourceRedshiftEvolution::prepareCandidate(Candidate& candidate) const {
858  double x = Random::instance().randUniform(0, 1);
859  double norm, z;
860 
861  // special case: m=-1
862  if ((std::abs(m+1)) < std::numeric_limits<double>::epsilon()) {
863  norm = log1p(zmax) - log1p(zmin);
864  z = exp(norm*x) * (1+zmin) - 1;
865  } else {
866  norm = ( pow(1+zmax, m+1) - pow(1+zmin, m+1) ) / (m+1);
867  z = pow( norm*(m+1)*x + pow(1+zmin, m+1), 1./(m+1)) - 1;
868  }
869  candidate.setRedshift(z);

889 SourceGenericComposition::SourceGenericComposition(double Emin, double Emax, std::string expression, size_t bins) :
890  Emin(Emin), Emax(Emax), expression(expression), bins(bins) {
891 
892  // precalculate energy bins
893  double logEmin = ::log10(Emin);
894  double logEmax = ::log10(Emax);
895  double logStep = (logEmax - logEmin) / bins;
896  energy.resize(bins + 1);

903 void SourceGenericComposition::add(int id, double weight) {
904  int A = massNumber(id);
905  int Z = chargeNumber(id);
906 
907  Nucleus n;
908  n.id = id;
909 
910  // calculate nuclei cdf
911  mu::Parser p;
912  double E;
913  p.DefineVar("E", &E);
914  p.DefineConst("Emin", Emin);
915  p.DefineConst("Emax", Emax);
916  p.DefineConst("bins", bins);
917  p.DefineConst("A", (double)A);
918  p.DefineConst("Z", (double)Z);
919 
920  p.DefineConst("MeV", MeV);
921  p.DefineConst("GeV", GeV);
922  p.DefineConst("TeV", TeV);
923  p.DefineConst("PeV", PeV);
924  p.DefineConst("EeV", EeV);
925 
926  p.SetExpr(expression);
927 
928  // calculate pdf
929  n.cdf.resize(bins + 1);
930 
931  for (std::size_t i=0; i<=bins; ++i) {
932  E = energy[i];
933  n.cdf[i] = p.Eval();
934  }
935 
936  // integrate
937  for (std::size_t i=bins; i>0; --i) {
938  n.cdf[i] = (n.cdf[i-1] + n.cdf[i]) * (energy[i] - energy[i-1]) / 2;
939  }
940  n.cdf[0] = 0;
941 
942  // cumulate
943  for (std::size_t i=1; i<=bins; ++i) {
944  n.cdf[i] += n.cdf[i-1];
945  }
946 
947  nuclei.push_back(n);
948 
949  // update composition cdf
950  if (cdf.size() == 0)
951  cdf.push_back(weight * n.cdf.back());
952  else
953  cdf.push_back(cdf.back() + weight * n.cdf.back());
954 }
