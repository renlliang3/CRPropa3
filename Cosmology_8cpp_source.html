29  void update() {
30  double dH = c_light / H0; // Hubble distance
31 
32  std::vector<double> E(n);
33  E[0] = 1;
34 
35  // Relation between comoving distance r and redshift z (cf. J.A. Peacock, Cosmological physics, p. 89 eq. 3.76)
36  // dr = c / H(z) dz, integration using midpoint rule
37  double dlz = log10(zmax) - log10(zmin);
38  for (int i = 1; i < n; i++) {
39  Z[i] = zmin * pow(10, i * dlz / (n - 1)); // logarithmic even spacing
40  double dz = (Z[i] - Z[i - 1]); // redshift step
41  E[i] = sqrt(omegaL + omegaM * pow_integer<3>(1 + Z[i]));
42  Dc[i] = Dc[i - 1] + dH * dz * (1 / E[i] + 1 / E[i - 1]) / 2;
43  Dl[i] = (1 + Z[i]) * Dc[i];
44  Dt[i] = Dt[i - 1]
45  + dH * dz
46  * (1 / ((1 + Z[i]) * E[i])
47  + 1 / ((1 + Z[i - 1]) * E[i - 1])) / 2;
48  }
49  }
