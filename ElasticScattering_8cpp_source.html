82 void ElasticScattering::process(Candidate *candidate) const {
83  int id = candidate->current.getId();
84  double z = candidate->getRedshift();
85 
86  if (not isNucleus(id))
87  return;
88 
89  double lg = log10(candidate->current.getLorentzFactor() * (1 + z));
90  if ((lg < lgmin) or (lg > lgmax))
91  return;
92 
93  int A = massNumber(id);
94  int Z = chargeNumber(id);
95  int N = A - Z;
96 
97  double step = candidate->getCurrentStep();
98  while (step > 0) {
99 
100  double rate = interpolateEquidistant(lg, lgmin, lgmax, tabRate);
101  rate *= Z * N / double(A); // TRK scaling
102  rate *= pow(1 + z, 2) * photonFieldScaling(photonField, z); // cosmological scaling
103 
104  // check for interaction
105  Random &random = Random::instance();
106  double randDist = -log(random.rand()) / rate;
107  if (step < randDist)
108  return;
109 
110  // draw random background photon energy from CDF
111  size_t i = floor((lg - lgmin) / (lgmax - lgmin) * (nlg - 1)); // index of closest gamma tabulation point
112  size_t j = random.randBin(tabCDF[i]) - 1; // index of next lower tabulated eps value
113  double binWidth = (epsmax - epsmin) / (neps - 1); // logarithmic bin width
114  double eps = pow(10, epsmin + (j + random.rand()) * binWidth);
115 
116  // boost to lab frame
117  double cosTheta = 2 * random.rand() - 1;
118  double E = eps * candidate->current.getLorentzFactor() * (1. - cosTheta);
119 
120  Vector3d pos = random.randomInterpolatedPosition(candidate->previous.getPosition(), candidate->current.getPosition());
121  candidate->addSecondary(22, E, pos);
122 
123  // repeat with remaining step
124  step -= randDist;
125  }
126 }
