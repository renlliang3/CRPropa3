34 void MagneticLens::loadLens(const string &filename)
35 {
36  ifstream infile(filename.c_str());
37  if (!infile)
38  {
39  throw std::runtime_error("Can't read file: " + filename);
40  }
41  string line;
42 
43  string prefix;
44  int sp = filename.find_last_of("/");
45  if (sp >= 0)
46  {
47  prefix = filename.substr(0, sp);
48  prefix.append("/");
49  }
50  string mfdatfile;
51  double emin, emax;
52 
53  int lineCounter = 0;
54  while (!infile.eof())
55  {
56  getline(infile, line);
57  lineCounter++;
58  if (line.find('#') == string::npos)
59  {
60  stringstream ss;
61  ss << line;
62  ss >> mfdatfile >> emin >> emax;
63  if (ss.fail())
64  {
65  cerr << "WARNING! Cannot read line " << lineCounter << " of file " << filename << " :\n [" << lineCounter << " ]: " << line << " \n";
66  cerr << " ... Skipping line and continue\n";
67  }
68  else
69  {
70  this->loadLensPart(prefix + mfdatfile, pow(10, emin) * eV, pow(10, emax) * eV);
71  }
72  }
73  }
74 }

76 bool MagneticLens::transformCosmicRay(double rigidity, double& phi,
77  double& theta)
78 {
79  uint32_t c = _pixelization->direction2Pix(phi, theta);
80  LensPart *lenspart = getLensPart(rigidity);
81  if (!lenspart)
82  {
83  std::cerr << "Warning. Trying to transform cosmic ray with rigidity " << rigidity / eV << " eV which is not covered by this lens!.\n";
84  std::cerr << " This lens covers the range " << _minimumRigidity /eV << " eV - " << _maximumRigidity << " eV.\n";
85  return false;
86  }
87 
88  ModelVectorType v = lenspart->getMatrix().col(c);
89 
90  uint32_t r;
91 
92  // the random number to compare with
93  double rn = Random::instance().rand();
94 
95  ModelVectorType::InnerIterator i(v);
96  double cpv = 0;
97  while (i)
98  {
99  cpv += i.value();
100  if (rn < cpv)
101  {
102  _pixelization->pix2Direction(i.index(), phi, theta);
103  return true;
104  }
105  else
106  {
107  ++i;
108  }
109  }
110  return false;
111 }

113 bool MagneticLens::transformCosmicRay(double rigidity, Vector3d &p){
114 
115  double galacticLongitude = atan2(-p.y, -p.x);
116  double galacticLatitude = M_PI / 2 - acos(-p.z/ sqrt(p.x*p.x + p.y*p.y + p.z*p.z));
117  bool result = transformCosmicRay(rigidity, galacticLongitude, galacticLatitude);
118 
119  p.x = -1 * cos(galacticLongitude) * sin(M_PI / 2 - galacticLatitude);
120  p.y = -1 * sin(galacticLongitude) * sin(M_PI / 2 - galacticLatitude);
121  p.z = -1. * cos(M_PI / 2 - galacticLatitude);
122 
123  return result;
124 }

138 void MagneticLens::_checkMatrix(const ModelMatrixType &M)
139 {
140  if (M.rows() != M.cols())
141  {
142  throw std::runtime_error("Not a square Matrix!");
143  }
144 
145  if (_pixelization)
146  {
147  if (_pixelization->nPix() != M.cols())
148  {
149  std::cerr << "*** ERROR ***" << endl;
150  std::cerr << " Pixelization: " << _pixelization->nPix() << endl;
151  std::cerr << " Matrix Size : " << M.cols() << endl;
152  throw std::runtime_error("Matrix doesn't fit into Lense");
153  }
154  }
155  else
156  {
157  uint32_t morder = Pixelization::pix2Order(M.cols());
158  if (morder == 0)
159  {
160  throw std::runtime_error(
161  "Matrix size doesn't match healpix scheme!");
162  }
163  _pixelization = new Pixelization(morder);
164  }

230 void MagneticLens::normalizeLens()
231 {
232  // get maximum of sums of columns, and normalize each matrix to that
233  double norm = 0;
234  for (LensPartIter iter = _lensParts.begin(); iter != _lensParts.end();
235  ++iter)
236  {

238  {
239  norm = (*iter)->getMaximumOfSumsOfColumns();

241  }
