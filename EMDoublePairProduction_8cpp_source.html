56 void EMDoublePairProduction::performInteraction(Candidate *candidate) const {
57  // the photon is lost in interaction
58  candidate->setActive(false);
59 
60  if (not haveElectrons)
61  return;
62 
63  // Use assumption of Lee 96 arXiv:9604098
64  // Energy is equally shared between one e+e- pair, but take mass of second e+e- pair into account.
65  // This approximation has been shown to be valid within -1.5%.
66  double E = candidate->current.getEnergy();
67  double Ee = (E - 2 * mass_electron * c_squared) / 2;
68 
69  Random &random = Random::instance();
70  Vector3d pos = random.randomInterpolatedPosition(candidate->previous.getPosition(), candidate->current.getPosition());
71 
72  candidate->addSecondary( 11, Ee, pos);
73  candidate->addSecondary(-11, Ee, pos);
74 }

76 void EMDoublePairProduction::process(Candidate *candidate) const {
77  // check if photon
78  if (candidate->current.getId() != 22)
79  return;
80 
81  // scale the electron energy instead of background photons
82  double z = candidate->getRedshift();
83  double E = (1 + z) * candidate->current.getEnergy();
84 
85  // check if in tabulated energy range
86  if (E < tabEnergy.front() or (E > tabEnergy.back()))
87  return;
88 
89  // interaction rate
90  double rate = interpolate(E, tabEnergy, tabRate);
91  rate *= pow(1 + z, 2) * photonFieldScaling(photonField, z);
92 
93  // check for interaction
94  Random &random = Random::instance();
95  double randDistance = -log(random.rand()) / rate;
96  if (candidate->getCurrentStep() > randDistance)
97  performInteraction(candidate);
98  else
99  candidate->limitNextStep(limit / rate);
100 }
