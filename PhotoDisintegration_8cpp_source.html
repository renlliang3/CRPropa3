148 void PhotoDisintegration::process(Candidate *candidate) const {
149  // execute the loop at least once for limiting the next step
150  double step = candidate->getCurrentStep();
151  do {
152  // check if nucleus
153  int id = candidate->current.getId();
154  if (not isNucleus(id))
155  return;
156 
157  int A = massNumber(id);
158  int Z = chargeNumber(id);
159  int N = A - Z;
160  size_t idx = Z * 31 + N;
161 
162  // check if disintegration data available
163  if ((Z > 26) or (N > 30))
164  return;
165  if (pdRate[idx].size() == 0)
166  return;
167 
168  // check if in tabulated energy range
169  double z = candidate->getRedshift();
170  double lg = log10(candidate->current.getLorentzFactor() * (1 + z));
171  if ((lg <= lgmin) or (lg >= lgmax))
172  return;
173 
174  double rate = interpolateEquidistant(lg, lgmin, lgmax, pdRate[idx]);
175  rate *= pow(1 + z, 2) * photonFieldScaling(photonField, z); // cosmological scaling, rate per comoving distance
176 
177  // check if interaction occurs in this step
178  // otherwise limit next step to a fraction of the mean free path
179  Random &random = Random::instance();
180  double randDist = -log(random.rand()) / rate;
181  if (step < randDist) {
182  candidate->limitNextStep(limit / rate);
183  return;
184  }
185 
186  // select channel and interact
187  const std::vector<Branch> &branches = pdBranch[idx];
188  double cmp = random.rand();
189  int l = round((lg - lgmin) / (lgmax - lgmin) * (nlg - 1)); // index of closest tabulation point
190  size_t i = 0;
191  while ((i < branches.size()) and (cmp > 0)) {
192  cmp -= branches[i].branchingRatio[l];
193  i++;
194  }
195  performInteraction(candidate, branches[i-1].channel);
196 
197  // repeat with remaining step
198  step -= randDist;
199  } while (step > 0);
200 }

202 void PhotoDisintegration::performInteraction(Candidate *candidate, int channel) const {
203  KISS_LOG_DEBUG << "Photodisintegration::performInteraction. Channel " << channel << " on candidate " << candidate->getDescription();
204  // parse disintegration channel
205  int nNeutron = digit(channel, 100000);
206  int nProton = digit(channel, 10000);
207  int nH2 = digit(channel, 1000);
208  int nH3 = digit(channel, 100);
209  int nHe3 = digit(channel, 10);
210  int nHe4 = digit(channel, 1);
211 
212  int dA = -nNeutron - nProton - 2 * nH2 - 3 * nH3 - 3 * nHe3 - 4 * nHe4;
213  int dZ = -nProton - nH2 - nH3 - 2 * nHe3 - 2 * nHe4;
214 
215  int id = candidate->current.getId();
216  int A = massNumber(id);
217  int Z = chargeNumber(id);
218  double EpA = candidate->current.getEnergy() / A;
219 
220  // create secondaries
221  Random &random = Random::instance();
222  Vector3d pos = random.randomInterpolatedPosition(candidate->previous.getPosition(), candidate->current.getPosition());
223  try
224  {
225  for (size_t i = 0; i < nNeutron; i++)
226  candidate->addSecondary(nucleusId(1, 0), EpA, pos);
227  for (size_t i = 0; i < nProton; i++)
228  candidate->addSecondary(nucleusId(1, 1), EpA, pos);
229  for (size_t i = 0; i < nH2; i++)
230  candidate->addSecondary(nucleusId(2, 1), EpA * 2, pos);
231  for (size_t i = 0; i < nH3; i++)
232  candidate->addSecondary(nucleusId(3, 1), EpA * 3, pos);
233  for (size_t i = 0; i < nHe3; i++)
234  candidate->addSecondary(nucleusId(3, 2), EpA * 3, pos);
235  for (size_t i = 0; i < nHe4; i++)
236  candidate->addSecondary(nucleusId(4, 2), EpA * 4, pos);
237 
238 
239  // update particle
240  candidate->created = candidate->current;
241  candidate->current.setId(nucleusId(A + dA, Z + dZ));
242  candidate->current.setEnergy(EpA * (A + dA));
243  }
244  catch (std::runtime_error &e)
245  {
246  KISS_LOG_ERROR << "Something went wrong in the PhotoDisentigration\n" << "Please report this error on https://github.com/CRPropa/CRPropa3/issues including your simulation setup and the following random seed:\n" << Random::instance().getSeed_base64();
247  throw;
248  }
249 
250  if (not havePhotons)
251  return;
252 
253  // create photons
254  double z = candidate->getRedshift();
255  double lg = log10(candidate->current.getLorentzFactor() * (1 + z));
256  double lf = candidate->current.getLorentzFactor();
257 
258  int l = round((lg - lgmin) / (lgmax - lgmin) * (nlg - 1)); // index of closest tabulation point
259  int key = Z*1e6 + (A-Z)*1e4 + (Z+dZ)*1e2 + (A+dA) - (Z+dZ);
260 
261  for (int i = 0; i < pdPhoton[key].size(); i++) {
262  // check for random emission
263  if (random.rand() > pdPhoton[key][i].emissionProbability[l])
264  continue;
265 
266  // boost to lab frame
267  double cosTheta = 2 * random.rand() - 1;
268  double E = pdPhoton[key][i].energy * lf * (1 - cosTheta);
269  candidate->addSecondary(22, E, pos);
270  }
271 }

273 double PhotoDisintegration::lossLength(int id, double gamma, double z) {
274  // check if nucleus
275  if (not (isNucleus(id)))
276  return std::numeric_limits<double>::max();
277 
278  int A = massNumber(id);
279  int Z = chargeNumber(id);
280  int N = A - Z;
281  size_t idx = Z * 31 + N;
282 
283  // check if disintegration data available
284  if ((Z > 26) or (N > 30))
285  return std::numeric_limits<double>::max();
286  const std::vector<double> &rate = pdRate[idx];
287  if (rate.size() == 0)
288  return std::numeric_limits<double>::max();
289 
290  // check if in tabulated energy range
291  double lg = log10(gamma * (1 + z));
292  if ((lg <= lgmin) or (lg >= lgmax))
293  return std::numeric_limits<double>::max();
294 
295  // total interaction rate
296  double lossRate = interpolateEquidistant(lg, lgmin, lgmax, rate);
297 
298  // comological scaling, rate per physical distance
299  lossRate *= pow(1 + z, 3) * photonFieldScaling(photonField, z);
300 
301  // average number of nucleons lost for all disintegration channels
302  double avg_dA = 0;
303  const std::vector<Branch> &branches = pdBranch[idx];
304  for (size_t i = 0; i < branches.size(); i++) {
305  int channel = branches[i].channel;
306  int dA = 0;
307  dA += 1 * digit(channel, 100000);
308  dA += 1 * digit(channel, 10000);
309  dA += 2 * digit(channel, 1000);
310  dA += 3 * digit(channel, 100);
311  dA += 3 * digit(channel, 10);
312  dA += 4 * digit(channel, 1);
313 
314  double br = interpolateEquidistant(lg, lgmin, lgmax, branches[i].branchingRatio);
315  avg_dA += br * dA;
316  }
317 
318  lossRate *= avg_dA / A;
319  return 1 / lossRate;
320 }
