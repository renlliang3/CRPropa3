179  T closestValue(const Vector3d &position) const {
180  Vector3d r = (position - gridOrigin) / spacing;
181  int ix = round(r.x);
182  int iy = round(r.y);
183  int iz = round(r.z);
184  if (reflective) {
185  while ((ix < 0) or (ix > Nx))
186  ix = 2 * Nx * (ix > Nx) - ix;
187  while ((iy < 0) or (iy > Ny))
188  iy = 2 * Ny * (iy > Ny) - iy;
189  while ((iz < 0) or (iz > Nz))
190  iz = 2 * Nz * (iz > Nz) - iz;
191  } else {
192  ix = ((ix % Nx) + Nx) % Nx;
193  iy = ((iy % Ny) + Ny) % Ny;
194  iz = ((iz % Nz) + Nz) % Nz;
195  }
196  return get(ix, iy, iz);
197  }

200  T interpolate(const Vector3d &position) const {
201  // position on a unit grid
202  Vector3d r = (position - gridOrigin) / spacing;
203 
204  // indices of lower and upper neighbors
205  int ix, iX, iy, iY, iz, iZ;
206  if (reflective) {
207  reflectiveClamp(r.x, Nx, ix, iX);
208  reflectiveClamp(r.y, Ny, iy, iY);
209  reflectiveClamp(r.z, Nz, iz, iZ);
210  } else {
211  periodicClamp(r.x, Nx, ix, iX);
212  periodicClamp(r.y, Ny, iy, iY);
213  periodicClamp(r.z, Nz, iz, iZ);
214  }
215 
216  // linear fraction to lower and upper neighbors
217  double fx = r.x - floor(r.x);
218  double fX = 1 - fx;
219  double fy = r.y - floor(r.y);
220  double fY = 1 - fy;
221  double fz = r.z - floor(r.z);
222  double fZ = 1 - fz;
223 
224  // trilinear interpolation (see http://paulbourke.net/miscellaneous/interpolation)
225  T b(0.);
226  //V000 (1 - x) (1 - y) (1 - z) +
227  b += get(ix, iy, iz) * fX * fY * fZ;
228  //V100 x (1 - y) (1 - z) +
229  b += get(iX, iy, iz) * fx * fY * fZ;
230  //V010 (1 - x) y (1 - z) +
231  b += get(ix, iY, iz) * fX * fy * fZ;
232  //V001 (1 - x) (1 - y) z +
233  b += get(ix, iy, iZ) * fX * fY * fz;
234  //V101 x (1 - y) z +
235  b += get(iX, iy, iZ) * fx * fY * fz;
236  //V011 (1 - x) y z +
237  b += get(ix, iY, iZ) * fX * fy * fz;
238  //V110 x y (1 - z) +
239  b += get(iX, iY, iz) * fx * fy * fZ;
240  //V111 x y z
241  b += get(iX, iY, iZ) * fx * fy * fz;
242 
243  return b;
244  }
