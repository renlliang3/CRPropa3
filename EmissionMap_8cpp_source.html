69  * Spherical Coordinates

73 size_t CylindricalProjectionMap::binFromDirection(const Vector3d& direction) const {
74  // convert to cylindrical
75  double phi = direction.getPhi() + M_PI;
76  double theta = sin(M_PI_2 - direction.getTheta()) + 1;
77 
78  // to indices
79  size_t iPhi = phi / sPhi;
80  size_t iTheta = theta / sTheta;
81 
82  // interleave
83  size_t bin = iTheta * nPhi + iPhi;
84  return bin;
85 }

87 Vector3d CylindricalProjectionMap::directionFromBin(size_t bin) const {
88  // deinterleave
89  double iPhi = bin % nPhi;
90  double iTheta = bin / nPhi;
91 
92  // any where in the bin
93  iPhi += Random::instance().rand();
94  iTheta += Random::instance().rand();
95 
96  // cylindrical Coordinates
97  double phi = iPhi * sPhi;
98  double theta = iTheta * sTheta;
99 
100  // sphericala Coordinates
101  phi = phi - M_PI;
102  theta = M_PI_2 - asin(theta - 1);
103 
104  Vector3d v;
105  v.setRThetaPhi(1.0, theta, phi);
106  return v;
107 }
