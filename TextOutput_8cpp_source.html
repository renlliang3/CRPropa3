277 void TextOutput::load(const std::string &filename, ParticleCollector *collector){
278 
279  std::string line;
280  std::istream *in;
281  std::ifstream infile(filename.c_str());
282 
283  double lengthScale = Mpc; // default Mpc
284  double energyScale = EeV; // default EeV
285 
286  if (!infile.good())
287  throw std::runtime_error("crpropa::TextOutput: could not open file " + filename);
288  in = &infile;
289 
290  if (kiss::ends_with(filename, ".gz")){
291 #ifdef CRPROPA_HAVE_ZLIB
292  in = new zstream::igzstream(*in);
293 #else
294  throw std::runtime_error("CRPropa was build without Zlib compression!");
295 #endif
296  }
297 
298  while (std::getline(*in,line)) {
299  std::stringstream stream(line);
300  if (stream.peek() == '#')
301  continue;
302 
303  ref_ptr<Candidate> c = new Candidate();
304  double val_d; int val_i;
305  double x, y, z;
306  stream >> val_d;
307  c->setTrajectoryLength(val_d*lengthScale); // D
308  stream >> val_d;
309  c->setRedshift(val_d); // z
310  stream >> val_i;
311  c->setSerialNumber(val_i); // SN
312  stream >> val_i;
313  c->current.setId(val_i); // ID
314  stream >> val_d;
315  c->current.setEnergy(val_d*energyScale); // E
316  stream >> x >> y >> z;
317  c->current.setPosition(Vector3d(x, y, z)*lengthScale); // X, Y, Z
318  stream >> x >> y >> z;
319  c->current.setDirection(Vector3d(x, y, z)*lengthScale); // Px, Py, Pz
320  stream >> val_i; // SN0 (TODO: Reconstruct the parent-child relationship)
321  stream >> val_i;
322  c->source.setId(val_i); // ID0
323  stream >> val_d;
324  c->source.setEnergy(val_d*energyScale); // E0
325  stream >> x >> y >> z;
326  c->source.setPosition(Vector3d(x, y, z)*lengthScale); // X0, Y0, Z0
327  stream >> x >> y >> z;
328  c->source.setDirection(Vector3d(x, y, z)*lengthScale); // P0x, P0y, P0z
329  stream >> val_i; // SN1
330  stream >> val_i;
331  c->created.setId(val_i); // ID1
332  stream >> val_d;
333  c->created.setEnergy(val_d*energyScale); // E1
334  stream >> x >> y >> z;
335  c->created.setPosition(Vector3d(x, y, z)*lengthScale); // X1, Y1, Z1
336  stream >> x >> y >> z;
337  c->created.setDirection(Vector3d(x, y, z)*lengthScale); // P1x, P1y, P1z
338  stream >> val_d;
339  c->setWeight(val_d); // W
340 
341  collector->process(c);
342  }
343  infile.close();
